Quick Start Guide

Using the Traced Package

To generate traces that can be viewed in this application, use the traced package with SQLite storage:

from traced import configure_tracing, Traced

# Configure SQLite storage
configure_tracing(
    storage_type="sqlite",
    database_path="/path/to/your/traces.db"  # Same path as configured in the viewer
)

# Create a traced class
class MyService(Traced):
    def process(self, data):
        # This method will be automatically traced
        result = self._transform(data)
        
        # Save an artifact
        self.save_artifact("processed_data", result, "json")
        
        return result
    
    def _transform(self, data):
        # Private methods are not traced
        return {"transformed": data}

# Use your class
service = MyService()
result = service.process({"input": "value"})
